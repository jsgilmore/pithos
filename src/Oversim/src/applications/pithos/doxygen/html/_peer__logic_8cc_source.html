00147 void Peer_logic::handleMessage(cMessage *msg)
00148 {
00149         if (msg == event)
00150         {
00151                 //For the first join request, a request is sent to the well known directory server
00152                 IPAddress dest_ip(directory_ip);
00153                 TransportAddress destAdr(dest_ip, directory_port);
00154 
00155                 joinRequest(destAdr);
00156 
00157                 scheduleAt(simTime()+1, event);         //TODO: make the 1 second wait time a configuration variable that may be set
00158         }
00159         else if (strcmp(msg->getArrivalGate()->getName(), "from_upperTier") == 0)
00160         {
00161                 //A storage request was received from the game or higher layer. This is data that should be stored in the network
00162                 handleRequest(msg);
00163         }
00164         else if (strcmp(msg->getArrivalGate()->getName(), "comms_gate$i") == 0)
00165         {
00166                 //Data was received from the UDP layer by the communicator and has been referred to the Peer logic
00167                 handleP2PMsg(msg);
00168                 delete(msg);
00169         }
00170         else {
00171                 error("Illegal message received");
00172                 delete(msg);
