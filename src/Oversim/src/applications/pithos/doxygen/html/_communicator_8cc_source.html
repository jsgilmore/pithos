00126 bool Communicator::handleRpcCall(BaseCallMessage *msg)
00127 {
00128 
00129     // There are many macros to simplify the handling of RPCs. The full list is in <OverSim>/src/common/RpcMacros.h.
00130 
00131     // start an RPC switch
00132     RPC_SWITCH_START(msg);
00133                 // RPCs between nodes
00134                 RPC_DELEGATE(DHTPut, handlePutRequest);
00135                 RPC_DELEGATE(DHTGet, handleGetRequest);
00136                 // internal RPCs
00137                 RPC_DELEGATE(DHTputCAPI, handlePutCAPIRequest);         //If we received a put request from Tier 2
00138                 RPC_DELEGATE(DHTgetCAPI, handleGetCAPIRequest);         //If we received a get request from Tier 2
00139                 RPC_DELEGATE(DHTdump, handleDumpDhtRequest);
00140     // end the switch
00141     RPC_SWITCH_END();
00142 
00143     // return whether we handled the message or not.
00144     // don't delete unhandled messages!
00145     return RPC_HANDLED;
